    <script type="text/javascript">
        // Input: Provide false, or blank string to ignore tracking
        var hstrack = window.hstrack = window.hstrack || {};
        hstrack.tracking_id = "C00116030";
        hstrack.ready = true;
        hstrack.queue.push(['trackPageView']);

        document.addEventListener('click', function(event) {
          var linkExtension = event.target.href.split('.').pop().toLowerCase();
          if (event.target.tagName === 'A' && linkExtension === 'pdf') { 
            hstrack.call(['trackEvent', {
              id: 'pdf_download',
              value: event.target.href
            }]);
          }
        });
    </script>
